  <script>

    let totalCredits = 0;

    let prizes = [

      { id: 1, name: "Extra Screen Time (30 min)", cost: 5 },

      { id: 2, name: "Choose Dinner", cost: 8 },

      { id: 3, name: "Ice Cream Treat", cost: 10 },

      { id: 4, name: "Toy from Store", cost: 25 }

    ];


    function loadData() {

      const saved = localStorage.getItem('credits') || '0';

      totalCredits = parseInt(saved);

      document.getElementById('creditCount').textContent = totalCredits;

      renderPrizes();

    }


    function saveCredits() {

      localStorage.setItem('credits', totalCredits);

      document.getElementById('creditCount').textContent = totalCredits;

    }


    function renderPrizes() {

      const container = document.getElementById('prizeList');

      container.innerHTML = '';

      

      prizes.forEach(prize => {

        const div = document.createElement('div');

        div.className = 'prize';

        div.innerHTML = `

          <span>${prize.name}</span>

          <span><strong>${prize.cost} credits</strong></span>

          <button class="redeem-btn" onclick="redeemPrize(${prize.id}, ${prize.cost})" 

            ${totalCredits >= prize.cost ? '' : 'disabled'}>

            Redeem

          </button>

        `;

        container.appendChild(div);

      });

    }


    function redeemPrize(id, cost) {

      if (totalCredits >= cost) {

        totalCredits -= cost;

        saveCredits();

        renderPrizes();

        alert("✅ Prize redeemed!");

      }

    }


    window.onload = loadData;

  </script>
