  <script>

    let stars = {};

    const hours = Array.from({length: 12}, (_, i) => 8 + i);

    const dailyGoal = 12;

    const creditsPerDay = 10;


    function loadData() {

      const saved = localStorage.getItem('starChart');

      stars = saved ? JSON.parse(saved) : {};

      renderGrid();

      checkDailyCompletion();

    }


    function saveData() {

      localStorage.setItem('starChart', JSON.stringify(stars));

      renderGrid();

      checkDailyCompletion();

    }


    function createShootingStar() {

      const star = document.createElement('div');

      star.className = 'shooting-star';

      const startX = Math.random() * window.innerWidth * 0.6;

      star.style.left = startX + 'px';

      star.style.top = Math.random() * (window.innerHeight * 0.4) + 'px';

      document.body.appendChild(star);

      setTimeout(() => star.remove(), 1500);

    }


    function toggleHour(hour) {

      const key = hour.toString();

      

      if (!stars) {

        stars = "earned";

        createShootingStar();

        setTimeout(createShootingStar, 180);

      } else if (stars === "earned") {

        stars = "lost";

      } else {

        stars = "earned";

        createShootingStar();

      }

      

      saveData();

    }


    function renderGrid() {

      const grid = document.getElementById('hourGrid');

      grid.innerHTML = '';

      

      hours.forEach(hour => {

        const div = document.createElement('div');

        div.className = `hour ${stars || ''}`;

        div.textContent = `${hour}:00`;

        div.onclick = () => toggleHour(hour);

        grid.appendChild(div);

      });


      const earned = Object.values(stars).filter(s => s === "earned").length;

      document.getElementById('starCount').textContent = `${earned} / ${dailyGoal}`;

    }


    function checkDailyCompletion() {

      const earned = Object.values(stars).filter(s => s === "earned").length;

      if (earned >= dailyGoal) {

        let credits = parseInt(localStorage.getItem('credits') || '0');

        if (!localStorage.getItem('dailyRewardGiven')) {

          credits += creditsPerDay;

          localStorage.setItem('credits', credits);

          localStorage.setItem('dailyRewardGiven', 'true');

          alert(`🎉 Daily goal reached! You earned ${creditsPerDay} credits!`);

        }

      }

    }


    function resetDay() {

      if (confirm("Reset all stars for today?")) {

        stars = {};

        localStorage.removeItem('dailyRewardGiven');

        saveData();

      }

    }


    window.onload = loadData;

  </script>
