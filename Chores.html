  <h1>🧹 Mom's Chore List 🧹</h1>

  <button onclick="addChore()">Add Chore</button>


  <div id="choreList"></div>


  <div class="nav-bar">

    <button class="nav-btn" onclick="window.location.href='index.html'">🌠</button>

    <button class="nav-btn" onclick="window.location.href='chores.html'">🧹</button>

    <button class="nav-btn" onclick="window.location.href='prizes.html'">🏆</button>

  </div>


  <script>

    let chores = [];


    function loadChores() {

      const saved = localStorage.getItem('chores');

      chores = saved ? JSON.parse(saved) : [

        { id: 1, text: "Clean your room", completed: false, value: 1 },

        { id: 2, text: "Take out the trash", completed: false, value: 1 }

      ];

      renderChores();

    }


    function saveChores() {

      localStorage.setItem('chores', JSON.stringify(chores));

      renderChores();

    }


    function addChore() {

      const input = document.getElementById('choreInput');

      if (!input.value.trim()) return;

      

      chores.push({

        id: Date.now(),

        text: input.value.trim(),

        completed: false,

        value: 1

      });

      

      input.value = "";

      saveChores();

    }


    function toggleChore(id) {

      const chore = chores.find(c => c.id === id);

      if (chore) {

        chore.completed = !chore.completed;

        saveChores();

      }

    }


    function deleteChore(id) {

      if (confirm("Delete this chore?")) {

        chores = chores.filter(c => c.id !== id);

        saveChores();

      }

    }


    function renderChores() {

      const container = document.getElementById('choreList');

      container.innerHTML = '';

      

      chores.forEach(chore => {

        const div = document.createElement('div');

        div.className = 'chore-item';

        div.innerHTML = `

          <span style="text-decoration:${chore.completed ? 'line-through' : 'none'}">

            ${chore.text}

          </span>

          <div>

            <button class="complete-btn" onclick="toggleChore(${chore.id})">

              ${chore.completed ? 'Undo' : 'Complete'}

            </button>

            <button class="delete-btn" onclick="deleteChore(${chore.id})">🗑️</button>

          </div>

        `;

        container.appendChild(div);

      });

    }


    window.onload = loadChores;

  </script>
